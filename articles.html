                void drawTorus() {
                    glColor3f(1.0, 1.0, 1.0);
                    for (int i = 0; i < TORUS_RINGS; i++) {
                        glBegin(GL_QUAD_STRIP);
                        for (int j = 0; j <= TORUS_SIDES; j++) {
                            for (int k = 0; k < 2; k++) {
                                float s = (i + k) % TORUS_RINGS + 0.5;
                                float t = j % (TORUS_SIDES + 1);
                
                                float x = (RADIUS_MAJOR + RADIUS_MINOR * cos(s * 2.0 * PI / TORUS_RINGS)) * cos(t * 2.0 * PI / TORUS_SIDES);
                                float y = (RADIUS_MAJOR + RADIUS_MINOR * cos(s * 2.0 * PI / TORUS_RINGS)) * sin(t * 2.0 * PI / TORUS_SIDES);
                                float z = RADIUS_MINOR * sin(s * 2.0 * PI / TORUS_RINGS);
                
                                glVertex3f(x, y, z);
                            }
                        }
                        glEnd();
                    }
                }
